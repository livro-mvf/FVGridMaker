Exemplos#

Os exemplos ativos têm nome ex_* e são compilados com BUILD_EXAMPLES=ON. Eles são exemplos curtos, usados para demonstrar a API sem alongar a leitura com comentários de manual.

Os programas completos e mais comentados ficam em Programas do manual. Eles seguem outro padrão: são material didático, com explicações no próprio código, verificações simples e saídas textuais voltadas à leitura no manual.

Compilar e executar#

Configure um build dedicado para exemplos:

cmake -S . -B build-examples -DBUILD_EXAMPLES=ON -DBUILD_TESTS=OFF

O projeto não permite ativar BUILD_TESTS e BUILD_EXAMPLES ao mesmo tempo. Essa separação mantém os testes e os exemplos em árvores de build distintas.

Para compilar todos os exemplos:

cmake --build build-examples -j

Para executar todos os exemplos:

cmake --build build-examples --target run_all_examples

Cada arquivo examples/.../ex_nome.cc ou examples/.../ex_nome.cpp gera um alvo executável ex_nome e um alvo de execução run_ex_nome. Por exemplo:

cmake --build build-examples --target ex_Axis1D
cmake --build build-examples --target run_ex_Axis1D

Os executáveis são gravados em:

build-examples/bin/examples/

 69struct FacesPonderadasDosCentros1D final {
 70    using primary_coordinates = fvgrid::CenterCoordinates1D;
 71
 72    [[nodiscard]] static constexpr std::string_view name() noexcept {
 73        return "FacesPonderadasDosCentros1D";
 74    }
 75
 76    [[nodiscard]] static fvgrid::AxisGeometry1D complete_geometry(
 77        std::vector<fvgrid::Real> centers,
 78        fvgrid::Domain1D domain
 79    ) {
 80        const fvgrid::Size nvol = centers.size();
 81
 82        std::vector<fvgrid::Real> faces(nvol + 1);
 83        faces.front() = domain.xmin();
 84        faces.back() = domain.xmax();
 85
 86        for (fvgrid::Size i = 1; i < nvol; ++i) {
 87            faces[i] = 0.25 * centers[i - 1] + 0.75 * centers[i];
 88        }
 89
 90        return fvgrid::AxisGeometry1D{
 91            .faces = std::move(faces),
 92            .centers = std::move(centers),
 93            .pattern_name = std::string{name()},
 94        };
 95    }
 96};

 98class MalhaSimetrica1D final {
 99public:
100    MalhaSimetrica1D(
101        fvgrid::Length comprimento,
102        fvgrid::NVol numero_de_volumes,
103        fvgrid::XInit x_inicial,
104        fvgrid::Seed semente
105    )
106        : comprimento_{comprimento},
107          numero_de_volumes_{numero_de_volumes},
108          x_inicial_{x_inicial},
109          semente_{semente} {}
110
111    [[nodiscard]] fvgrid::Axis1D gerar() const {
112        std::vector<fvgrid::Real> centros = gerar_centros();
113
114        return fvgrid::Custom1D::make(
115            fvgrid::Coordinates1D::centers(std::move(centros)),
116            FacesPonderadasDosCentros1D{},
117            fvgrid::Domain1D::from_length(x_inicial_, comprimento_)
118        );
119    }
120
121private:
122    [[nodiscard]] std::vector<fvgrid::Real> gerar_centros() const {
123        const fvgrid::Size n = numero_de_volumes_.value();
124        const fvgrid::Real xmin = x_inicial_.value();
125        const fvgrid::Real xmax = xmin + comprimento_.value();
126        const fvgrid::Real xmeio = 0.5 * (xmin + xmax);
127
128        const fvgrid::Size metade = n / 2;
129
130        std::mt19937_64 gerador{semente_.value()};
131        std::uniform_real_distribution<fvgrid::Real> distribuicao{xmin, xmeio};
132
133        std::vector<fvgrid::Real> centros_esquerda(metade);
134
135        for (fvgrid::Size i = 0; i < metade; ++i) {
136            centros_esquerda[i] = distribuicao(gerador);
137        }
138
139        std::sort(centros_esquerda.begin(), centros_esquerda.end());
140
141        std::vector<fvgrid::Real> centros(n);
142
143        for (fvgrid::Size i = 0; i < metade; ++i) {
144            centros[i] = centros_esquerda[i];
145        }
146
147        if (n % 2 != 0) {
148            centros[metade] = xmeio;
149        }
150
151        for (fvgrid::Size i = 0; i < metade; ++i) {
152            centros[n - 1 - i] = 2.0 * xmeio - centros_esquerda[i];
153        }
154
155        return centros;
156    }
157
158    fvgrid::Length comprimento_;
159    fvgrid::NVol numero_de_volumes_;
160    fvgrid::XInit x_inicial_;
161    fvgrid::Seed semente_;
162};

164[[nodiscard]] bool registrar_teste(
165    const std::string& descricao,
166    bool passou
167) {
168    std::cout << (passou ? "[PASSOU] " : "[FALHOU] ")
169              << descricao << '\n';

173[[nodiscard]] bool centros_simetricos(
174    const fvgrid::Axis1D& malha,
175    fvgrid::Real tolerancia
176) {
177    const auto centros = malha.centers();
178    const fvgrid::Real xmeio = 0.5 * (malha.xmin() + malha.xmax());
179    bool passou = true;
180
181    for (fvgrid::Size i = 0; i < centros.size(); ++i) {
182        const fvgrid::Size j = centros.size() - 1 - i;
183
184        if (i > j) {
185            break;
186        }
187
188        const fvgrid::Real residual =
189            std::abs((centros[i] + centros[j]) - 2.0 * xmeio);
190
191        if (residual > tolerancia) {
192            std::cout << "  centros[" << i << "] + centros[" << j
193                      << "] difere de 2*xmeio por "
194                      << residual << '\n';
195            passou = false;
196        }
197    }
198
199    return passou;
200}
201
202[[nodiscard]] bool faces_ponderadas_corretas(
203    const fvgrid::Axis1D& malha,
204    fvgrid::Real tolerancia
205) {
206    const auto faces = malha.faces();
207    const auto centros = malha.centers();
208    bool passou = true;
209
210    for (fvgrid::Size i = 1; i + 1 < faces.size(); ++i) {
211        const fvgrid::Real face_esperada =
212            0.25 * centros[i - 1] + 0.75 * centros[i];
213
214        const fvgrid::Real residual = std::abs(faces[i] - face_esperada);
215
216        if (residual > tolerancia) {
217            std::cout << "  face[" << i
218                      << "] nao obedece a regra 25/75; residuo = "
219                      << residual << '\n';
220            passou = false;
221        }
222    }
223
224    return passou;
225}

243void imprimir_razao_de_espacamentos(const fvgrid::Axis1D& malha) {
244    const auto dx = malha.dx_faces();
245    const auto [min_it, max_it] = std::minmax_element(dx.begin(), dx.end());

247    std::cout << "dx_max/dx_min = " << (*max_it / *min_it) << "\n\n";
248}

250[[nodiscard]] unsigned verificar_malha(
251    const fvgrid::Axis1D& malha,
252    fvgrid::Real tolerancia
253) {
254    unsigned aprovados{};
255    unsigned total{};
256
257    std::cout << "\nVerificacoes automaticas\n";
258    std::cout << "========================\n";
259
260    ++total;
261    aprovados += registrar_teste(
262        "centros simetricos em relacao ao ponto medio",
263        centros_simetricos(malha, tolerancia)
264    );
265
266    ++total;
267    aprovados += registrar_teste(
268        "faces internas obedecem a regra 25/75",
269        faces_ponderadas_corretas(malha, tolerancia)
270    );
271
272    std::cout << "\nResumo\n";
273    std::cout << "======\n";
274    std::cout << "Verificacoes aprovadas: "
275              << aprovados << " de " << total << '\n';
276
277    return (aprovados == total) ? 0u : 1u;
278}

 98class MalhaSimetrica1D final {
 99public:
100    MalhaSimetrica1D(
101        fvgrid::Length comprimento,
102        fvgrid::NVol numero_de_volumes,
103        fvgrid::XInit x_inicial,
104        fvgrid::Seed semente
105    )
106        : comprimento_{comprimento},
107          numero_de_volumes_{numero_de_volumes},
108          x_inicial_{x_inicial},
109          semente_{semente} {}
110
111    [[nodiscard]] fvgrid::Axis1D gerar() const {
112        std::vector<fvgrid::Real> centros = gerar_centros();
113
114        return fvgrid::Custom1D::make(
115            fvgrid::Coordinates1D::centers(std::move(centros)),
116            FacesPonderadasDosCentros1D{},
117            fvgrid::Domain1D::from_length(x_inicial_, comprimento_)
118        );
119    }

121private:
122    [[nodiscard]] std::vector<fvgrid::Real> gerar_centros() const {
123        const fvgrid::Size n = numero_de_volumes_.value();
124        const fvgrid::Real xmin = x_inicial_.value();
125        const fvgrid::Real xmax = xmin + comprimento_.value();
126        const fvgrid::Real xmeio = 0.5 * (xmin + xmax);
127
128        const fvgrid::Size metade = n / 2;
129
130        std::mt19937_64 gerador{semente_.value()};
131        std::uniform_real_distribution<fvgrid::Real> distribuicao{xmin, xmeio};
132
133        std::vector<fvgrid::Real> centros_esquerda(metade);
134
135        for (fvgrid::Size i = 0; i < metade; ++i) {
136            centros_esquerda[i] = distribuicao(gerador);
137        }
138
139        std::sort(centros_esquerda.begin(), centros_esquerda.end());
140
141        std::vector<fvgrid::Real> centros(n);
142
143        for (fvgrid::Size i = 0; i < metade; ++i) {
144            centros[i] = centros_esquerda[i];
145        }
146
147        if (n % 2 != 0) {
148            centros[metade] = xmeio;
149        }
150
151        for (fvgrid::Size i = 0; i < metade; ++i) {
152            centros[n - 1 - i] = 2.0 * xmeio - centros_esquerda[i];
153        }
154
155        return centros;
156    }

160    fvgrid::XInit x_inicial_;
161    fvgrid::Seed semente_;

173[[nodiscard]] bool centros_simetricos(
174    const fvgrid::Axis1D& malha,
175    fvgrid::Real tolerancia
176) {
177    const auto centros = malha.centers();
178    const fvgrid::Real xmeio = 0.5 * (malha.xmin() + malha.xmax());
179    bool passou = true;
180
181    for (fvgrid::Size i = 0; i < centros.size(); ++i) {
182        const fvgrid::Size j = centros.size() - 1 - i;
183
184        if (i > j) {
185            break;
186        }
187
188        const fvgrid::Real residual =
189            std::abs((centros[i] + centros[j]) - 2.0 * xmeio);
190
191        if (residual > tolerancia) {
192            std::cout << "  centros[" << i << "] + centros[" << j
193                      << "] difere de 2*xmeio por "
194                      << residual << '\n';
195            passou = false;
196        }
197    }
198
199    return passou;
200}

202[[nodiscard]] bool faces_ponderadas_corretas(
203    const fvgrid::Axis1D& malha,
204    fvgrid::Real tolerancia
205) {
206    const auto faces = malha.faces();
207    const auto centros = malha.centers();
208    bool passou = true;
209
210    for (fvgrid::Size i = 1; i + 1 < faces.size(); ++i) {
211        const fvgrid::Real face_esperada =
212            0.25 * centros[i - 1] + 0.75 * centros[i];
213
214        const fvgrid::Real residual = std::abs(faces[i] - face_esperada);
215
216        if (residual > tolerancia) {
217            std::cout << "  face[" << i
218                      << "] nao obedece a regra 25/75; residuo = "
219                      << residual << '\n';
220            passou = false;
221        }
222    }
223
224    return passou;
